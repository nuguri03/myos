# 1. "Hello BootLoader!" 출력하기

## 📖 소스 코드
```asm
[ORG 0x7c00]        // 메모리의 시작부분은 0x7c00로, 그 전 부분은 인터럽트 벡터 등 BIOS에서 사용할 부분
[BITS 16]           // nasm이 16비트로 컴파일하도록

start:
    mov si, msg     // si 레지스터 = 문자열 인덱스의 첫번째 부분을 가르킬 때 사용하는 레지스터
    call print
    jmp $           // 중간에 종료 되지 않게 무한 반복

print:
    mov ah, 0x0e    // ah = ax(16비트)의 윗부분, 즉 8비트 부분 / 0xoe = 출력 메뉴판 준비
.loop:
    lodsb           // 문자열을 읽고 si레지스터 값 1개씩 증가
    cmp al, 0
    je .end         // je = jump equal  
    int 0x10        // 출력을 일으키는 인터럽트
    jmp .loop
.end:
    ret             // return
    
msg db 'Hello BootLoader!', 0       // db = 1BYTE

times 510-($-$$) db 0               // $ = 현재 메모리 위치, $$ = 처음 시작한 메모리 위치
dw 0xAA55                           // dw = 2BYTE
```
